def remove_bias(df):
    mu = np.mean(df - df.mean(), axis=1)
    
    ratings_df = pd.DataFrame()
    for i, row in enumerate(df.iterrows()):
        ratings_df = ratings_df.append(pd.Series(row[1] - mu[i]), ignore_index=True) 
    return ratings_df.mean()

def box_plots(path, title):
    data_tr, data_lol = read_and_clean_data(path)
    
    sns.swarmplot(data=data_tr[['tr-66_','tr-74_', 'tr-78_','tr-82_', 'tr-86_','tr-92_','tr-94_',  'tr-orig_']], color='black')
    sns.boxplot(data=data_tr[['tr-66_','tr-74_', 'tr-78_','tr-82_', 'tr-86_','tr-92_','tr-94_',  'tr-orig_']])
    plt.title('Tomb Raider - {}'.format(title))
    plt.xlabel('Rendition')
    plt.ylabel('Score')
    plt.show()
    
    sns.swarmplot(data=data_lol[['lol-75_', 'lol-78_', 'lol-82_', 'lol-87_', 'lol-92_', 'lol-96_', 'lol-orig_']], color='black')
    sns.boxplot(data=data_lol[['lol-75_', 'lol-78_', 'lol-82_', 'lol-87_', 'lol-92_', 'lol-96_', 'lol-orig_']])
    plt.title('League of Legends - {}'.format(title))
    plt.xlabel('Rendition')
    plt.ylabel('Score')
    plt.show()
